int binarySearch(int arr[], int l, int r, int x)  // Recursive 
{ 
    if (r >= l) 
    { 
        int mid = l + (r - l) / 2; 
  
        if (arr[mid] == x) 
            return mid; 
   
        if (arr[mid] > x) 
            return binarySearch(arr, l, mid - 1, x); 
  
        return binarySearch(arr, mid + 1, r, x); 
    } 
  
    // We reach here when element is not present in array
    return -1; 
}
